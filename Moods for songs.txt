HIGH ENERGY
select track_name, artists.artist_name, songs.genre_num,song_genre, energy from songs, attributes, artists, genre where songs.song_num = attributes.song_num and attributes.genre_num = genre.genre_num and genre.artist_name = artists.artist_name and energy > '.9';

high Energy, valence
